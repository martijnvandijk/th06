INCLUDE(CMakeForceCompiler)

SET(CMAKE_SYSTEM_NAME Linux)
SET(CMAKE_SYSTEM_VERSION 1)
SET(CMAKE_SYSTEM_PROCESSOR arm)
SET(CMAKE_CROSSCOMPILING TRUE)
MESSAGE(STATUS "Cross compiling for ${CMAKE_SYSTEM_PROCESSOR}")

SET(GCC_PREFIX ${CMAKE_CURRENT_LIST_DIR}/bin/arm-linux-gnueabihf-)

SET(CMAKE_C_COMPILER ${GCC_PREFIX}gcc)
SET(CMAKE_CXX_COMPILER ${GCC_PREFIX}g++)

CMAKE_FORCE_C_COMPILER(${GCC_PREFIX}gcc GNU)
CMAKE_FORCE_CXX_COMPILER(${GCC_PREFIX}g++ GNU)

SET(CMAKE_FIND_ROOT_PATH ${CMAKE_CURRENT_LIST_DIR}/rpi-rootfs/)
SET(CMAKE_FIND_ROOT_PATH_MODE_PROGRAM NEVER)
SET(CMAKE_FIND_ROOT_PATH_MODE_LIBRARY ONLY)
SET(CMAKE_FIND_ROOT_PATH_MODE_INCLUDE ONLY)
